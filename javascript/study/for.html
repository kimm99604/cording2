반복문

반복문은 프로그램에서 필요한 결과 값을 도출하기 위해 실행문의 순서를 반복적으로 실행시키는 문법을 말합니다. 반복문에는 while문, do while문, for문이 있으며, for문을 가장 많이 사용합니다.

while문

while문은 반복할 때 사용하는 문법이다.

초기값
while (조건식) {
     실행문;
     증가감식;
}

do while문

* do wile 문은 do를 먼저 시랳ㅇ하기 때문에 첫 번째 실행은 조건에 상관없이 실행합니다.

초기값
do {
     실행문;
     증가감식;
} while (조건식)

for문

for문도 if문처럼 for 문 안에 또 다른 for문을 중첩하여 사용할 수 있습니다.

for(초기 값; 조건식; 증가감식)
     실행문;
}

0부터 99까지 출력하기

//0부터 99까지 출력하기
for( let i=0 ; i<100; i++){
    document.write(i);
}

for문을 이용해서 테이블 출력하기

중첩 for문을 이용해서 테이블을 만들 수 있습니다.

//for문을 이용해서 테이블 출력하기(100칸)
let table = "<table>";

for(let i=0; i<10; i++){
    table += "<tr>";

    for(let j=0; j<10; j++){
        table += "<td>*</td>";
    }

    table += "</tr>";
}
table += "</table>";

document.write(table);

for문을 이용해서 1-100까지 출력(5의배수, 7의배수)

for문을 이용해 5의 배수는 파란색, 7의 배수는 빨간색으로 표시한다.

for( let i=1; i<=100; i++ ){
    if( i % 5 == 0 ){ // 5로 나누면 0으로 떨어짐 
        document.write("<span style='color:blue'>"+i+"</span>");
    }
    else if( i % 7 == 0 ){ // 7로 나누면 0으로 떨어짐 
        document.write("<span style='color:red'>"+i+"</span>");
    }
}
결과 확인하기

for문을 이용해서 1-100까지 출력(짝수는 빨간색, 홀수는 파란색)

짝수는 빨간색으로 표시하고 홀수는 파란색으로 표시한다.

for( let i=1; i<=100; i++ ){

for문을 이용해서 1-100까지 출력(짝수는 빨간색, 홀수는 파란색) : 조건부 연산자

짝수는 빨간색으로 표시하고 홀수는 파란색으로 표시한다.

//조건문 ? true : false
for(let i=1; i<=100; i++){
    (i % 2 == 0) ? document.write("<span style='color:red'>" + i + "</span>") : document.write("<span style='color:blue'>" + i + "</span>");
}

구구단 출력하기

for문을 통해 2-9단까지 구구단을 만들 수 있다.

for(let i=2; i<=9; i++){
    for(let j=1; j<=9; j++){
        let sum = i*j;
        document.write(i+ " * " + j + " = " + sum);

for문을 통해 표 만들기

100칸의 표를 만들 수 있다.

let table = "<table class='testTable'>";
let num = 0;

for(let i=1; i<=10; i++){     
    table += "<tr>";
    
    for(let j=1; j<=10; j++){
        num++;
        table += "<td>"+num+"</td>";
    }  
    table += "</tr>"; 
}   

table += "</tr>"; 
table += "<table>";
document.write(table);
