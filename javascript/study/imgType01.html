CSS

transform: scale로 사진을 확대하고 transition으로 시간차 효과를 주었다. 

/* fonts */
@import url('https://webfontworld.github.io/gmarket/GmarketSans.css');

.nexon {
    font-family: 'GmarketSans';
    font-weight: 300;
}

/* reset */
* {
    padding: 0;
    margin: 0;
}
a {
    text-decoration: none;
    color: #000;
}
img {
    width: 100%;
}

/* common */
.container {
    width: 1160px;
    padding: 0 20px;
    margin: 0 auto;
    min-width: 1160px;
}
.section {
    padding: 120px 0;
}
.section > h2 {
    font-size: 50px;
    line-height: 1; /*line height auto는 1을 의미*/
    text-align: center;
    margin-bottom: 20px;
}
.section > p {
    font-size: 22px;
    font-weight: 300;
    color: #666;
    text-align: center;
    margin-bottom: 70px;
}
h1, h2, h3, h4, h5 ,h6 {
    font-weight: normal;
}

    vertical-align: top;
}

    position: absolute;
    left: 0;
    bottom: -100px;
    color: #fff;

    text-align: center;

    box-sizing: border-box;
    font-size: 24px;
    background-color: rgba(0, 0, 0, 0.089);
