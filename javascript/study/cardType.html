CSS

flex와 space-between을 통해 가로 정렬, 여백을 맞추고 card width를 32%로 주어 동일한 사이즈의 카드를 3개 만들었습니다. 

/* fonts */
    @import url('https://webfontworld.github.io/NexonLv1Gothic/NexonLv1Gothic.css');

    .nexon {
        font-family: "NexonLv1Gothic";
        font-weight: 400;
    }

    /* reset */
    * {
        padding: 0;
        margin: 0;
    }
    a {
        text-decoration: none;
        color: #000;
    }
    img {
        width: 100%;
    }

    /* common */
    .container {
        width: 1160px;
        padding: 0 20px;
        margin: 0 auto;
    }
    .section {
        padding: 120px 0;
    }
    .section > h2 {
        font-size: 50px;
        line-height: 1; /*line height auto는 1을 의미*/
        text-align: center;
        margin-bottom: 20px;
    }
    .section > p {
        font-size: 22px;
        font-weight: 300;
        color: #666;
        text-align: center;
        margin-bottom: 70px;
    }
    h1, h2, h3, h4, h5 ,h6 {
        font-weight: normal;
    }

    /* cardType */
    .card_inner {
        display: flex;
        justify-content: space-between
    }

        width: 32%;
        background-color: #f5f5f5;
    }
