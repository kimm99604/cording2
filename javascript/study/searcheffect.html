html, css, js 태그&메서드 검색 reference

원본 소스 보기

JavaScript

"keyup" 이벤트를 통해 사용자가 입력한 키워드를 검색할 수 있게 만들었습니다. 

<script>
const searchBox = document.querySelector(".search__box input"); //검색 박스
const searchList = document.querySelectorAll(".search__list ul li"); //리스트

searchBox.addEventListener("keyup", () => {
    const searchWord = searchBox.value; //사용자가 입력한키워드
    searchList.forEach((el) => {
        const cssName = el.dataset.name;
        if (cssName.indexOf(searchWord)) {
            el.classList.add("hide");
        } else {
            el.classList.remove("hide"); 
        }
    });
})
</script>

검색 이펙트 03

html, css, js - 알파벳별 목록 리스트

결과

원본 소스 보기

JavaScript

이번엔 charAt 메서드를 통해 알파벳별 목록 리스트를 만들었습니다. 

<script>
    const searchList = document.querySelectorAll(".search__list ul li"); //리스트
        const searchInfo = document.querySelector(".num"); //전체 갯수 
        const searchKeyword = document.querySelectorAll(".search__info .keyword span"); //키워드 

        //전체 갯수 구하기 
        searchInfo.textContent = searchList.length;
       

        //알파벳별 검색 함수 
        searchKeyword.forEach(el => { //el=span
            el.addEventListener("click", ()=> {
                const searchWord = el.innerText; //사용자가 클릭한 알파벳 

                console.log(searchWord)

                searchList.forEach(el => {
                    const cssName = el.querySelector("strong").innerText; //css 속성 테스트
                    // console.log(cssName)

                    //알파벳 첫글자 == CSS 속성의 첫글자 
                    if(searchWord.charAt(0) === cssName.charAt(0)){
                        el.classList.remove("hide");
                    } else {
                        el.classList.add("hide");
                    }
                });
            });
        });
</script>
