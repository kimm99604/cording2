함수(function)

함수는 프로그램에서 반복적으로 사용되는 기능을 만들어 내기 위한 코드들의 집합

선언적 함수

'선언적 함수'는 함수를 호출할 때 함수의 앞 또는 뒤에서 호출 가능합니다.

선언적 함수 형식

function 함수()) {
     실행문;
} 

//01. 선언적 함수
function func() {
    document.write("함수가 실행되었습니다.1");
}
func(); // 함수를 꼭 호출해줘야 실행됨

익명 함수

익명함수는 변수에 함수 데이터를 저장하여 변수를 마치 함수처럼 사용할 수 있도록 만들어줍니다.

익명 함수 형식

var 변수 = function() {
     실행문;
} 

//02. 익명 함수
const func = function() {
    document.write("함수가 실행되었습니다.2");
}
func();

매개변수 함수

매개변수로 나타내는 함수입니다.

매개변수 함수 형식

function 함수명(매개변수1,2,...)) {
     실행문;
}
함수명(매개변수1 값, 매개변수 2 값....); 

 //03. 매개변수(parameter) 함수
//변수 : 지역변수, 전역변수, 매개변수
function func(str){     //str은 매개변수임 
    document.write(str);
}
func("함수가 실행되었습니다.3");

리턴값 함수

함수 안에서 return을 만나게 되면 해당 함수를 호출한 곳으로 결과 데이터를 반환해 주고 함수는 종료가 됩니다.

리턴값 함수 형식

function 함수명() {
     실행문;
     return;
}
함수호출; 

//04. 리턴값 함수
function func(){
    const str = "함수가 실행되었습니다.4";
    return str; //리턴값에 str문자열 저장 /실행문이 있으면 return생략가능 
}
document.write(func()); //return값이 있기때문에 document에 함수 선언가능

화살표 함수

'=>'을 이용하여 함수를 간결하게 표현할 때 사용합니다. (es6문법)

화살표 함수 형식

함수명 = () => {
     실행문;
}
함수호출; 

//화살표 함수 
//1.
function func(str){
    document.write(str);
}
func("함수가 실행되었습니다.");

//2. 
func = (str) => {
    document.write(str);
}
func("함수가 실행되었습니다.");

//3. 
func = (str) => document.write(str);
func("함수가 실행되었습니다.");

//4. 앞에 const 생략가능 
func = str => document.write(str);
func("함수가 실행되었습니다.");
